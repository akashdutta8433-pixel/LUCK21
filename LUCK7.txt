<script>
let time=30,balance=5000;
let bets={number:[],color:[],bs:[]};
let history=[];

// create numbers
const nums=document.getElementById("nums");
for(let i=0;i<=9;i++){
  let b=document.createElement("button");
  b.innerText=i;
  b.onclick=()=>toggle('number',i,b);
  nums.appendChild(b);
}

// timer
setInterval(()=>{
  time--;
  if(time<=0){time=30;result();}
  document.getElementById("time").innerText=time;
},1000);

function toggle(type,val,btn){
  if(bets[type].includes(val)){
    bets[type]=bets[type].filter(x=>x!==val);
    btn.classList.remove("active");
  }else{
    bets[type].push(val);
    btn.classList.add("active");
  }
  showSel();
}

function showSel(){
  let t=[];
  bets.number.length&&t.push("Number:"+bets.number.join(","));
  bets.color.length&&t.push("Color:"+bets.color.join(","));
  bets.bs.length&&t.push("BS:"+bets.bs.join(","));
  sel.innerText=t.join(" | ")||"None";
}

function result(){
  let amt=+betAmount.value;
  if(amt<1||amt>10000||amt>balance){reset();return;}

  let n=Math.floor(Math.random()*10);
  let c=n%2==0?"Green":"Red";
  let s=n<=4?"Small":"Big";
  let win=0,loss=0;

  bets.number.length&&(bets.number.includes(n)?win+=amt*9:loss+=amt);
  bets.color.forEach(x=>x==c?win+=amt*2:loss+=amt);
  bets.bs.forEach(x=>x==s?win+=amt*2:loss+=amt);

  balance+=win-loss;
  bal.innerText=balance;
  res.innerText=`${n} (${c}, ${s}) | +₹${win} -₹${loss}`;

  history.unshift(
    `Bet ₹${amt} | ${sel.innerText} → ${n}(${c},${s}) | +₹${win} -₹${loss}`
  );
  history=history.slice(0,10);
  betHistory.innerHTML=history.join("<hr>");

  reset();
}

function reset(){
  bets={number:[],color:[],bs:[]};
  document.querySelectorAll("button").forEach(b=>b.classList.remove("active"));
  betAmount.value="";
  showSel();
}

const sel=document.getElementById("sel");
const res=document.getElementById("res");
const bal=document.getElementById("bal");
const betHistory=document.getElementById("betHistory");
</script>
